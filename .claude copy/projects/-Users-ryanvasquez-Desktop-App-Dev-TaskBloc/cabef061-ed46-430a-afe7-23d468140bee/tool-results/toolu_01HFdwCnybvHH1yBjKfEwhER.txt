   446→function SyncStatus({status}){
   447→  const [online,setOnline]=React.useState(navigator.onLine);
   448→  React.useEffect(()=>{
   449→    const on=()=>setOnline(true);
   450→    const off=()=>setOnline(false);
   451→    window.addEventListener('online',on);
   452→    window.addEventListener('offline',off);
   453→    return()=>{window.removeEventListener('online',on);window.removeEventListener('offline',off);};
   454→  },[]);
   455→  if(!online)return(<div className="sync-bar"><div className="sync-dot err"/><span>offline — changes will sync when reconnected</span></div>);
   456→  const cls=status==='ok'?'ok':status==='error'?'err':status==='syncing'?'syncing':'';
   457→  const label=status==='ok'?'synced':status==='error'?'sync error':status==='syncing'?'syncing…':'connecting…';
   458→  return(<div className="sync-bar"><div className={'sync-dot '+cls}/><span>{label}</span></div>);
   459→}

   482→  const toggleTask=(k,id)=>setTasks(p=>{const next={...p,[k]:(p[k]||[]).map(t=>t.id===id?{...t,complete:!t.complete}:t)};scheduleSave(next,forward);return next;});
   483→  const saveTask=(k,u)=>setTasks(p=>{const next={...p,[k]:(p[k]||[]).map(t=>t.id===u.id?u:t)};scheduleSave(next,forward);return next;});

   485→  const delTask=(k,id)=>{dbDeleteTask(id);setTasks(p=>{const next={...p,[k]:(p[k]||[]).filter(t=>t.id!==id)};scheduleSave(next,forward);return next;});setEditing(null);};

   487→  const moveToFwd=(fd,id)=>{const task=(tasks[fd]||[]).find(t=>t.id===id);if(!task)return;setTasks(prev=>{const next={...prev,[fd]:(prev[fd]||[]).filter(t=>t.id!==id)};const nf=[...forward,task];scheduleSave(next,nf);setForward(nf);return next;});};
   488→  const addFwd=n=>{const task={id:uid(),name:n,status:'',context:'',complete:false,attachments:[]};setForward(prev=>{const next=[...prev,task];scheduleSave(tasks,next);return next;});};

   491→  const delFwd=id=>{dbDeleteForward(id);setForward(p=>{const next=p.filter(t=>t.id!==id);scheduleSave(tasks,next);return next;});setEditFwd(null);};
