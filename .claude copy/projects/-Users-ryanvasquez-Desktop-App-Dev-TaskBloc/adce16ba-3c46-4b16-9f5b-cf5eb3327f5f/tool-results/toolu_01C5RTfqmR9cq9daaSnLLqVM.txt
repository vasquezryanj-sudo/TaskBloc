   398→      if(op.type==='upsertForward')return sb.from('forward_tasks').upsert(op.data);
   399→      if(op.type==='deleteForward')return sb.from('forward_tasks').delete().eq('id',op.id);
   400→      if(op.type==='upsertNextWeek')return sb.from('next_week').upsert(op.data);
   401→      if(op.type==='deleteNextWeek')return sb.from('next_week').delete().eq('id',op.id);
   402→    }));
   403→  }catch(e){
   404→    // restore queue if flush fails
   405→    const remaining=qLoad();
   406→    qSave([...q,...remaining]);
   407→    throw e;
   408→  }
   409→}
