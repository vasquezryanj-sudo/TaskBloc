    93→function FolderPicker({
    94→  book,
    95→  allBooks,
    96→  onAssign,
    97→}: {
    98→  book: Book
    99→  allBooks: Book[]
   100→  onAssign: (bookId: string, folderName: string | null) => Promise<void>
   101→}) {
   102→  const [open, setOpen] = useState(false)
   103→  const [creating, setCreating] = useState(false)
   104→  const [newName, setNewName] = useState('')
   105→  const ref = useRef<HTMLDivElement>(null)
   106→  const inputRef = useRef<HTMLInputElement>(null)
   107→
   108→  const existingFolders = Array.from(
   109→    new Set(allBooks.map(b => b.folder_name).filter((f): f is string => !!f))
   110→  ).sort((a, b) => a.localeCompare(b))
   111→
   112→  useEffect(() => {
   113→    if (!open) return
   114→    function handleClick(e: MouseEvent) {
   115→      if (ref.current && !ref.current.contains(e.target as Node)) {
   116→        setOpen(false)
   117→        setCreating(false)
   118→        setNewName('')
   119→      }
   120→    }
   121→    function handleKey(e: KeyboardEvent) {
   122→      if (e.key === 'Escape') {
   123→        setOpen(false)
   124→        setCreating(false)
   125→        setNewName('')
   126→      }
   127→    }
   128→    document.addEventListener('pointerdown', handleClick)
   129→    document.addEventListener('keydown', handleKey)
   130→    return () => {
   131→      document.removeEventListener('pointerdown', handleClick)
   132→      document.removeEventListener('keydown', handleKey)
   133→    }
   134→  }, [open])
   135→
   136→  useEffect(() => {
   137→    if (creating) inputRef.current?.focus()
   138→  }, [creating])
   139→
   140→  function handleSelect(folder: string) {
   141→    onAssign(book.id, folder)
   142→    setOpen(false)
   143→    setCreating(false)
   144→    setNewName('')
   145→  }
   146→
   147→  function handleCreate() {
   148→    const trimmed = newName.trim()
   149→    if (!trimmed) return
   150→    onAssign(book.id, trimmed)
   151→    setOpen(false)
   152→    setCreating(false)
   153→    setNewName('')
   154→  }
   155→
   156→  return (
   157→    <div ref={ref} className="relative shrink-0">
   158→      <button
   159→        onClick={e => { e.stopPropagation(); setOpen(o => !o) }}
   160→        className="w-7 h-7 flex items-center justify-center rounded-full text-fog-400 hover:text-navy-700 hover:bg-fog-100 transition-all"
   161→        title="Assign folder"
   162→      >
   163→        <svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 20 20" fill="currentColor" className="w-3.5 h-3.5">
   164→          <path d="M3.75 3A1.75 1.75 0 0 0 2 4.75v3.26a3.235 3.235 0 0 1 1.75-.51h12.5c.644 0 1.245.188 1.75.51V6.75A1.75 1.75 0 0 0 16.25 5h-4.836a.25.25 0 0 1-.177-.073L9.823 3.513A1.75 1.75 0 0 0 8.586 3H3.75ZM3.75 9A1.75 1.75 0 0 0 2 10.75v4.5c0 .966.784 1.75 1.75 1.75h12.5A1.75 1.75 0 0 0 18 15.25v-4.5A1.75 1.75 0 0 0 16.25 9H3.75Z" />
   165→        </svg>
   166→      </button>
   167→
   168→      {open && (
   169→        <div
   170→          className="absolute right-0 top-full mt-1 w-52 bg-white rounded-xl border border-fog-300 shadow-lg z-50 py-1 overflow-hidden"
   171→          onClick={e => e.stopPropagation()}
   172→        >
   173→          {/* New folder */}
   174→          {creating ? (
   175→            <div className="px-3 py-2 flex items-center gap-1.5">
   176→              <input
   177→                ref={inputRef}
   178→                type="text"
   179→                value={newName}
   180→                onChange={e => setNewName(e.target.value)}
   181→                onKeyDown={e => { if (e.key === 'Enter') handleCreate() }}
   182→                placeholder="Folder name"
   183→                className="flex-1 min-w-0 text-sm px-2 py-1 rounded-lg border border-fog-300 focus:border-navy-400 focus:outline-none"
   184→              />
   185→              <button
   186→                onClick={handleCreate}
   187→                disabled={!newName.trim()}
   188→                className="text-xs font-medium text-navy-700 hover:text-navy-900 px-2 py-1 rounded-lg hover:bg-fog-100 transition-all disabled:opacity-30"
   189→              >
   190→                Add
   191→              </button>
   192→            </div>
   193→          ) : (
   194→            <button
   195→              onClick={() => setCreating(true)}
   196→              className="w-full text-left px-3 py-2 text-sm text-navy-700 hover:bg-fog-100 transition-colors"
   197→            >
   198→              Add to new folder…
   199→            </button>
   200→          )}
   201→
   202→          {/* Existing folders */}
   203→          {existingFolders.length > 0 && (
   204→            <>
   205→              <div className="border-t border-fog-200 my-1" />
   206→              {existingFolders.map(folder => (
   207→                <button
   208→                  key={folder}
   209→                  onClick={() => handleSelect(folder)}
   210→                  className={`w-full text-left px-3 py-2 text-sm hover:bg-fog-100 transition-colors truncate ${
   211→                    book.folder_name === folder ? 'text-amber-700 font-medium' : 'text-navy-800'
   212→                  }`}
   213→                >
   214→                  {folder}
   215→                </button>
   216→              ))}
   217→            </>
   218→          )}
   219→        </div>
   220→      )}
   221→    </div>
   222→  )
   223→}

   227→function SortableBookRow({
   228→  book,
   229→  onOpenDetail,
   230→  onComplete,
   231→  onMoveUp,
   232→  showEmoji,
   233→  allBooks,
   234→  onAssignFolder,
   235→}: {
   236→  book: Book
   237→  onOpenDetail: () => void
   238→  onComplete: (book: Book) => void
   239→  onMoveUp?: () => void
   240→  showEmoji?: boolean

   245→    id: book.id,

   250→    transition,

   268→        {...listeners}
   269→        className="text-fog-300 hover:text-fog-500 cursor-grab active:cursor-grabbing shrink-0 touch-none transition-colors"
   270→        aria-label="Drag to reorder"
   271→      >

   277→        onClick={onOpenDetail}

   298→            </div>
   299→            <span className="text-xs text-fog-400">
   300→              p.{book.current_page ?? 0}/{book.total_pages}
   301→            </span>
   302→          </div>
   303→        )}

   305→
   306→      {/* Folder picker */}
   307→      <FolderPicker book={book} allBooks={allBooks} onAssign={onAssignFolder} />
   308→
   309→      {/* Action */}

   311→        <button
   312→          onClick={e => { e.stopPropagation(); onMoveUp() }}

   316→        </button>
   317→      ) : (
   318→        <CompleteButton onClick={() => onComplete(book)} />
   319→      )}
   320→    </div>
   321→  )
   322→}

   501→function typeEmoji(type: Book['type']): string {
   502→  if (type === 'article') return '📰'
   503→  if (type === 'podcast') return '🎙️'
   504→  return '📚'
   505→}
   506→
   507→function CompletedMediaSection({
   508→  books,
   509→  articles,
   510→  podcasts,
   511→  onOpenDetail,
   512→}: {
   513→  books: Book[]
   514→  articles: Book[]
   515→  podcasts: Book[]
   516→  onOpenDetail: (book: Book) => void
   517→}) {
   518→  const router = useRouter()
   519→
   520→  function sortRecent(items: Book[]) {
   521→    return [...items].sort((a, b) =>
   522→      (b.date_finished ?? b.created_at ?? '').localeCompare(a.date_finished ?? a.created_at ?? '')
   523→    )
   524→  }
   525→
   526→  const recentBooks = sortRecent(books).slice(0, 2)
   527→  const recentArticles = sortRecent(articles).slice(0, 2)
   528→  const recentPodcasts = sortRecent(podcasts).slice(0, 2)
   529→  const combined = [...recentBooks, ...recentArticles, ...recentPodcasts]
   530→
   531→  if (combined.length === 0) return null
   532→
   533→  return (
   534→    <section>
   535→      <h2 className="text-[11px] font-bold uppercase tracking-widest text-[#cc5500] mb-3">
   536→        Complete
   537→      </h2>
   538→      <div className="space-y-1.5">
   539→        {combined.map(book => (
   540→          <button
   541→            key={book.id}
   542→            onClick={() => onOpenDetail(book)}
   543→            className="w-full text-left bg-white rounded-xl border border-fog-300 shadow-sm px-4 py-3 flex items-center gap-3 hover:border-fog-400 hover:shadow-md transition-all group"
   544→          >
   545→            <div className="flex-1 min-w-0">
   546→              <p className="text-sm font-medium text-navy-800 truncate group-hover:text-navy-900">
   547→                {typeEmoji(book.type)} {book.title}
   548→              </p>
   549→              {book.author && (

   766→  async function handleDragEnd(event: DragEndEvent) {
   767→    const { active, over } = event
   768→    setActiveDragBookId(null)
   769→    if (!over) return
   770→
   771→    const activeId = active.id as string
   772→    const overId = over.id as string
   773→    if (activeId === overId) return
   774→
   775→    const inReading = readingNow.some(b => b.id === activeId)
   776→    const inQueue = nextUp.some(b => b.id === activeId)
   777→
   778→    const overIsReadingSection = overId === READING_SECTION
   779→    const overIsQueueSection = overId === QUEUE_SECTION
   780→    const overInReading = readingNow.some(b => b.id === overId)
   781→    const overInQueue = nextUp.some(b => b.id === overId)
   782→
   783→    let destSection: 'reading' | 'queue' | null = null
   784→    if (overIsReadingSection || overInReading) destSection = 'reading'
   785→    else if (overIsQueueSection || overInQueue) destSection = 'queue'
   786→    if (!destSection) return
   787→
   788→    const sourceSection = inReading ? 'reading' : inQueue ? 'queue' : null
   789→    if (!sourceSection) return
   790→
   791→    if (sourceSection === destSection) {
   792→      const list = sourceSection === 'reading' ? readingNow : nextUp
   793→      const oldIndex = list.findIndex(b => b.id === activeId)
   794→      const newIndex = (overIsReadingSection || overIsQueueSection)
   795→        ? list.length - 1
   796→        : list.findIndex(b => b.id === overId)
   797→      if (oldIndex === -1 || newIndex === -1 || oldIndex === newIndex) return
   798→
   799→      const reordered = arrayMove(list, oldIndex, newIndex)
   800→      setBooks(prev => {
   801→        const others = prev.filter(b => !list.some(l => l.id === b.id))
   802→        return [...others, ...reordered.map((b, i) => ({ ...b, sort_order: i + 1 }))]
   803→      })
   804→      await fetch('/api/books/reorder', {
   805→        method: 'PATCH',
   806→        headers: { 'Content-Type': 'application/json' },
   807→        body: JSON.stringify({ orderedIds: reordered.map(b => b.id) }),
   808→      })
   809→      await refresh()
   810→    } else {
   811→      if (destSection === 'reading' && allReadingNow.length >= 10) return
   812→      const originalStatus = books.find(b => b.id === activeId)?.status
   813→      const newStatus = destSection === 'reading' ? 'active' : 'queued'
   814→
   815→      setBooks(prev => prev.map(b =>
   816→        b.id === activeId ? { ...b, status: newStatus } : b
   817→      ))
   818→
   819→      const res = await fetch(`/api/books/${activeId}`, {
   820→        method: 'PATCH',
   821→        headers: { 'Content-Type': 'application/json' },
   822→        body: JSON.stringify({ status: newStatus }),
   823→      })
   824→      if (!res.ok && originalStatus) {
   825→        setBooks(prev => prev.map(b =>
   826→          b.id === activeId ? { ...b, status: originalStatus } : b
   827→        ))
   828→      }
   829→    }
   830→  }
